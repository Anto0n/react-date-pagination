## Install

```bash
npm install --save react-date-pagination
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-date-pagination'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
